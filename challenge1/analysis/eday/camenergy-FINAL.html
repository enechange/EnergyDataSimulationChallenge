#function to calculate MAPE
def mean_absolute_percentage_error(y_true, y_pred):
    y_true, y_pred = np.array(y_true), np.array(y_pred)
    return np.mean(np.abs((y_true - y_pred) / y_true)) * 100

house_mean = train.groupby('House')['EnergyProduction'].mean()
house_mean_sd = train.groupby('House')['EnergyProduction'].std()
train = train.join(house_mean, on='House', rsuffix='_house_mean')
train = train.join(house_mean_sd, on='House', rsuffix='_sd')
train['House_Mean_std'] = (train['EnergyProduction'] - train['EnergyProduction_house_mean'])/train['EnergyProduction_sd']
train.drop(['EnergyProduction_house_mean', 'EnergyProduction_house_mean'], 1)

lookup_base = dict(zip(house_mean.index, house_mean))
